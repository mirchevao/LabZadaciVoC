Од СВ се вчитува квадратна матрица А[nxn]. Да се трансформира матрицата на следниов начин: ако елементот над споредната дијагонала е поголем од неговиот
симетричен елемент во однос на споредната дијагонала, тогаш тие два елементи треба да си ги заменат местата, а во спротивно остануваат на истите места.

#include <stdio.h>
int main()
{
    int pom,n,i,b,j,a[100][100];
    scanf("%d",&n);
    b=n-1;
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
        	scanf("%d",&a[i][j]);
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
           if(i+j<b)
                if((a[i][j]>a[b-j][b-i]))
                {
                   pom=a[i][j];
                   a[i][j]=a[b-j][b-i];
                   a[b-j][b-i]=pom;
                        
                }
   
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
      		printf("%d ",a[i][j]);
        printf("\n");
    }
    
    
            
      return 0;
}
