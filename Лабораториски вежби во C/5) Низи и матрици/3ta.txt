За дадена матрица која се вчитува од СВ да се испечатат сите елементи кои се наоѓаат над и под главната дијагонала и над и под споредната дијагонала.

#include <stdio.h>
#define MAX 100

int main() {
	// vchituvanje na matrica
    int m, n;
    scanf("%d %d", &m, &n);
    int a[MAX][MAX];
    int i, j;
    for(i = 0; i < m; i++) {
        for(j = 0; j < n; j++) {
            scanf("%d", &a[i][j]);
        }
    }
    
    // vashiot kod ovde
    
    
    return 0;
    
}
