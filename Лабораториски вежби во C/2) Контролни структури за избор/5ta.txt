Да се напише програма во која од СВ се чита ден и месец, а потоа се печати на СИ кое годишно време е во тој датум.

# include <stdio.h>
int main(){

     int d,m;
    scanf("%d %d",&d,&m);
    
    if(((m==3)&&(d<=22))||(m==4)||(m==5)||((m==6)&&(d<=22)))
        printf("prolet");
    if(((m==6)&&(d>=22))||(m==7)||(m==8)||((m==9)&&(d<=22)))
        printf("leto");
    if(((m==9)&&(d>=23))||(m==10)||(m==11)||((m==12)&&(d<=21)))
        printf("esen");
    if(((m==12)&&(d>=22))||(m==1)||(m==2)||((m==3)&&(d<=22)))
        printf("zima");
       return 0;
}
