Да се напише програма во коjа за даден опсег на природни броеви од a до b (се читаат СВ) ке ги оптечати на СИ сите броеви 
чии што збир на цифри кои се деливи со 3 е поголем од 0 и е делив со 4 . 
На крај ќе го отпечати вкупниот број на вакви броеви.

#include <stdio.h>
int main()
{
    int i,a,b,a1,a2,c,suma=0,d;
    scanf("%d",&a);
    scanf("%d",&b);
    for(i=a;i<=b;i++)
    {
        suma=0;
        c=i;
        while(c!=0)
        {
            a2=c%10;
            if(a2%3==0)
            {
                suma+=a2;
            }
            c=c/10;
        }
        if ((suma>0)&&(suma%4==0))
        {
            printf("%d\n",i);
            d++;
        }
    }
    printf("%d",d);
    return 0;
}
