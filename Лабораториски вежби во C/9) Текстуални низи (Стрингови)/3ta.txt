int stringMatch(char *str, char *match) {

    b=strlen(match);
    a=strlen(str);
    for(i=0;i<a;i++)
    {
        for(j=0;j<b;j++)
        {
            if(str[i+j]!=match[j])

                break;
            }
        }
        if(j==b)
        {
            brojac++;
        }
    }

}
